<a name="сутьсистемы"></a>
### Суть системы
После нажатия на бинд OfferItem на F, у передающего появляется оверлей OfferItemIndicatorsOverlay, переключая в OfferItemComponent IsInOfferMode. После клика по тому, кому надо передать предмет, у клиента появляется алерт. При клике на алерт предмет передается IsInOfferMode становится на офф

<a name="архитектура"></a>
## Архитектура
```
Схема потока данных:

[Игрок нажимает на F]
        │
        ▼
[Shared: SharedOfferItemSystem.Interactions.SetInOfferMode()]
   └── Если до этого были какие либо данные в компоненте, обнуляем, если еще не в оффер моде - переводим
   
        │
        ▼
[Shared: SharedOfferItemSystem.SetInRecieveMode()]
   └── За счет InteractUsingEvent, который вызывается при клике в оффер моде по сущности, заполняем Target в компоненте вставляя в него сущность цель, выводим из оффер мода если тот был у цели 
   
        │
        ▼
[Server: OfferItemSystem.Update()]
   ├── каждый кадр проверяем таймер
   ├── Если есть рука и нету предмета = обнуляем все и отменяем оффер
   └── Выдаем алерт или убираем его если в режиме получения или нет 
        │
        ▼
[Shared: SharedOfferItemSystem.OnClickAlertEvent()]
   └── При клике на алерт переходим в метод Recieve
        │
        ▼
[Shared: SharedOfferItemSystem.Recieve()]
   └── После проверок передаем предмет и переходим в UnRecieve
        │
        ▼
[Shared: SharedOfferItemSystem.Recieve()]
   └── После проверок обнуляем все и отменяем передачу
```

<a name="сервер"></a>
## Server
Своеобразный менеджер, каждый кадр, после таймера проверяем сущности на то, есть ли у них руки и так далее. Если нету, обнуляем компоненты оффера, потом же или показываем (если offerItem.IsInReceiveMode=true) или же очищаем алерт оффера.

```
    public override void Update(float frameTime)
    {
        _offerAcc += frameTime;

        if (_offerAcc >= OfferAccMax)
            _offerAcc -= OfferAccMax;

        else
            return;
		// Таймер между передачами

        var query = EntityQueryEnumerator<OfferItemComponent, HandsComponent>();

        while (query.MoveNext(out var uid, out var offerItem, out var hands))
        {
            if (hands.ActiveHandId is null) 
                continue;
			// Проверка на то, есть ли активная рука

            if (offerItem.Hand is not null && _hands.GetActiveItem(uid) is null) // проверка на то, есть ли рука и есть ли активная рука

                if (offerItem.Target is not null)
                {
                    UnReceive(offerItem.Target.Value, offerItem: offerItem);
                    offerItem.IsInOfferMode = false;
                    Dirty(uid, offerItem);
                    // Если таргет не нулевой, отзываем получение и переводим оффер мод в фолс
                }

                else
                    UnOffer(uid, offerItem);
                    // отзываем оффер

            if (!offerItem.IsInReceiveMode)
            {
                _alertsSystem.ClearAlert(uid, OfferAlert);
                continue;
                // Если не в режиме получения очищаем алерт
            }

            _alertsSystem.ShowAlert(uid, OfferAlert);
            // показываем алерт в противном случае
        }
    }
}
```

```
    private void SetInOfferMode(ICommonSession? session)
    {
        if (!_timing.IsFirstTimePredicted)
            return;

        if (session is null)
            return;
// /\
// | проверки
        if (session.AttachedEntity is not { Valid: true } uid || !Exists(uid) || !_actionBlocker.CanInteract(uid, null)) // Если ентити недоступна или вообще не существует и не может интерактировать - возращаем
            return;

        if (!TryComp<OfferItemComponent>(uid, out var offerItem)) // проверяем есть ли оффер компонент у сущности и извлекаем сам компач при присутствии
            return;

        if (!TryComp<HandsComponent>(uid, out var hands) || hands.ActiveHandId is null) // Проверяем, есть ли у сущности рука
            return;

        offerItem.Item = _hand.GetActiveItem((uid, hands));

        if (!offerItem.IsInOfferMode)
        {
            if (offerItem.Item is null)
            {
                _popup.PopupClient(Loc.GetString("offer-item-empty-hand"), uid, uid); // Если рука пуста - отправляем попап с сообщением, что рука пуста
                return;
            }

            if (offerItem.Hand is null || offerItem.Target is null)
            {
                offerItem.IsInOfferMode = true;
                offerItem.Hand = hands.ActiveHandId;

                Dirty(uid, offerItem);
                return;
                // заполняем компонент если данные компонента пусты
            }
        }

        if (offerItem.Target is not null)
        {
            UnReceive(offerItem.Target.Value, offerItem: offerItem);
            offerItem.IsInOfferMode = false;
            Dirty(uid, offerItem);
            return;
            // прерываем передачу, если уже есть данные
        }

        UnOffer(uid, offerItem);
    }
```

Проверки и перемещение в руки с попапом предмета
```
    public void Receive(Entity<OfferItemComponent?> ent)
    {
        if (!_timing.IsFirstTimePredicted)
            return;

        if (!Resolve(ent, ref ent.Comp))
            return;

        if (!TryComp<OfferItemComponent>(ent.Comp.Target, out var offerItem)) // проверка на компонент
            return;

        if (offerItem.Hand is null) // если не существует рука, в которой держим предмет - возращаем
            return;

        if (ent.Comp.Target is null) если не существует того, кому передаем предмет - возращаем
            return;

        if (!TryComp<HandsComponent>(ent, out var hands)) // проверка на руку
            return;

        if (offerItem.Item is not null)
        {
            if (!_hand.TryPickup(ent, offerItem.Item.Value, handsComp: hands))
            {
                _popup.PopupClient(Loc.GetString("offer-item-full-hand"), ent, ent); // если руки полные - выводим попап об этом
                return;
            }

            _popup.PopupClient(Loc.GetString("offer-item-give",
                ("item", Identity.Entity(offerItem.Item.Value, EntityManager)),
                ("target", Identity.Entity(ent, EntityManager))), ent.Comp.Target.Value, ent.Comp.Target.Value); // если усе норм - выдаем попап для таргета

            _popup.PopupPredicted(Loc.GetString("offer-item-give-other",
                    ("user", Identity.Entity(ent.Comp.Target.Value, EntityManager)),
                    ("item", Identity.Entity(offerItem.Item.Value, EntityManager)),
                    ("target", Identity.Entity(ent, EntityManager)))
                , ent.Comp.Target.Value, ent); // если усе норм - выдаем попап для юзера
        }

        offerItem.Item = null;
        Dirty(ent);
        UnReceive(ent, ent, offerItem);
        // очищаем компоненты
    }
```

Отменяем офферы и ресетаем все значения с попапами
```
    protected void UnOffer(EntityUid uid, OfferItemComponent component)
    {
        if (!TryComp<HandsComponent>(uid, out var hands) || hands.ActiveHandId is null) // проверяем есть ли руки и есть ли айди руки в компаче
            return;

        if (TryComp<OfferItemComponent>(component.Target, out var offerItem) && component.Target is not null) // проверка на компач оффера и есть ли таргет в компаче
        {
            if (component.Item is not null)
            {
                if (!_timing.IsFirstTimePredicted)
                {
                    _popup.PopupClient(Loc.GetString("offer-item-no-give",
                        ("item", Identity.Entity(component.Item.Value, EntityManager)),
                        ("target", Identity.Entity(component.Target.Value, EntityManager))), uid, uid); // Вызываем попап который оповещает, что оффер отменен
                    _popup.PopupClient(Loc.GetString("offer-item-no-give-target",
                        ("user", Identity.Entity(uid, EntityManager)),
                        ("item", Identity.Entity(component.Item.Value, EntityManager))), uid, component.Target.Value); // Вызываем попап который оповещает, что оффер отменен таргету
                }

            }
            else if (offerItem.Item is not null)
                if (!_timing.IsFirstTimePredicted)
                {
                    _popup.PopupClient(Loc.GetString("offer-item-no-give",
                        ("item", Identity.Entity(offerItem.Item.Value, EntityManager)),
                        ("target", Identity.Entity(uid, EntityManager))), component.Target.Value, component.Target.Value); // Вызываем попап который оповещает, что оффер отменен
                    _popup.PopupClient(Loc.GetString("offer-item-no-give-target",
                        ("user", Identity.Entity(component.Target.Value, EntityManager)),
                        ("item", Identity.Entity(offerItem.Item.Value, EntityManager))), component.Target.Value, uid);// Вызываем попап который оповещает, что оффер отменен таргету
                }

            offerItem.IsInOfferMode = false;
            offerItem.IsInReceiveMode = false;
            offerItem.Hand = null;
            offerItem.Target = null;
            offerItem.Item = null;

            Dirty(component.Target.Value, offerItem);
            // ресетаем значения в компоненте
        }

        component.IsInOfferMode = false;
        component.IsInReceiveMode = false;
        component.Hand = null;
        component.Target = null;
        component.Item = null;

        Dirty(uid, component);
        // ресетаем значения в компоненте
    }
```

Проверки и ануление всех переменных
```
    protected void UnReceive(EntityUid uid, OfferItemComponent? component = null, OfferItemComponent? offerItem = null)
    {
        if (component is null && !TryComp(uid, out component)) // проверка на компонент
            return;

        if (offerItem is null && !TryComp(component.Target, out offerItem)) // проверка на айтем и компонент
            return;

        if (!TryComp<HandsComponent>(uid, out var hands) || hands.ActiveHandId is null ||
            component.Target is null) // проверка есть ли руки и есть ли активная рука
            return;

        if (offerItem.Item is not null)
        {
            _popup.PopupClient(Loc.GetString("offer-item-no-give",
                ("item", Identity.Entity(offerItem.Item.Value, EntityManager)),
                ("target", Identity.Entity(uid, EntityManager))), component.Target.Value, component.Target.Value);
            _popup.PopupClient(Loc.GetString("offer-item-no-give-target",
                ("user", Identity.Entity(component.Target.Value, EntityManager)),
                ("item", Identity.Entity(offerItem.Item.Value, EntityManager))), component.Target.Value, uid); // попапы те же, что и в Аноффер
        }

        if (!offerItem.IsInReceiveMode)
        {
            offerItem.Target = null;
            component.Target = null;
            // если не в режиме получения онуляемм значения компонентов
        }

        offerItem.Item = null;
        offerItem.Hand = null;
        component.IsInReceiveMode = false;

        Dirty(uid, component);
        // онуляем знаечения компонентов
    }
```

```
    protected override bool BeforeDraw(in OverlayDrawArgs args)
    {
        return _offer.IsInOfferMode() && base.BeforeDraw(in args);
    }

    protected override void Draw(in OverlayDrawArgs args)
    {
        var mousePosMap = _eye.PixelToMap(_inputManager.MouseScreenPosition);

        if (mousePosMap.MapId != args.MapId)
            return;

        var limitedScale = Math.Min(1.25f, (args.ViewportControl as Control)?.UIScale ?? 1f) * _scale;

        DrawSight(_sight, args.ScreenHandle, _inputManager.MouseScreenPosition.Position, limitedScale);
    }

    private void DrawSight(Texture sight, DrawingHandleScreen screen, Vector2 centerPos, float scale)
    {
        var sightSize = sight.Size * scale;
        var expandedSize = sightSize + new Vector2(7);

        screen.DrawTextureRect(sight, UIBox2.FromDimensions(centerPos - sightSize * 0.5f, sightSize), _strokeColor);
        screen.DrawTextureRect(sight, UIBox2.FromDimensions(centerPos - expandedSize * 0.5f, expandedSize), _mainColor);
    }
```
